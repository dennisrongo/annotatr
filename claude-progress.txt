2. **Hotkey Registration (lib.rs:560-620)**
```rust
#[tauri::command]
fn register_hotkeys(app: AppHandle, hotkey_config: serde_json::Value) -> Result<(), String> {
    unregister_all_hotkeys(app.clone())?;

    for (hotkey_name, hotkey_value) in hotkeys_config.as_object().unwrap() {
        let shortcut = Shortcut::new(hotkey_value.clone())?;
        let tool = hotkey_name.clone();

        app.global_shortcut().register(shortcut, move || {
            if let Err(e) = activate_tool_hotkey(app_handle.clone(), tool.clone()) {
                eprintln!("Failed to activate tool '{}': {}", tool, e);
            }
        })?;
    }
}
```

3. **Tool Activation (lib.rs:861-894)**
```rust
#[tauri::command]
fn activate_tool_hotkey(app: AppHandle, tool: String) -> Result<(), String> {
    let tool_type = convert_hotkey_tool_name(&tool);

    // Show overlay if not visible
    if !overlay_window.is_visible()? {
        overlay_window.show()?;
        overlay_window.set_focus()?;
        overlay_window.set_always_on_top(true)?;
        state_guard.drawing_mode = true;
    }

    // Emit tool-selected event to overlay
    app.emit("tool-selected", tool_type)?;
    app.emit("drawing-mode-changed", true)?;

    Ok(())
}
```

The application uses `tauri-plugin-global-shortcut v2`, which provides a single, unified API for
registering global hotkeys across Windows, macOS, and Linux. This is the industry-standard approach

on macOS, and X11/Wayland directly on Linux. This would require three separate code paths,
increasing complexity and bug surface area.
