Key Implementation Points:

State Variables:
// Hotkey customization state
const [hotkeys, setHotkeys] = useState<Record<string, string>>(DEFAULT_SETTINGS.hotkeys);
const [editingHotkey, setEditingHotkey] = useState<string | null>(null);
const [capturedHotkey, setCapturedHotkey] = useState<string>("");
Hotkey Capture Function:
const handleHotkeyCapture = (e: React.KeyboardEvent) => {
  if (!editingHotkey) return;
  e.preventDefault();

  const keys: string[] = [];

  // Capture modifier keys
  if (e.ctrlKey) keys.push("Ctrl");
  if (e.shiftKey) keys.push("Shift");
  if (e.altKey) keys.push("Alt");
  if (e.metaKey || e.key === "Meta")
    keys.push(isMac() ? "Cmd" : "Win");

  // Capture the main key
  const mainKey = e.key;
  if (!["Control", "Shift", "Alt", "Meta"].includes(mainKey)) {
    if (mainKey.length === 1) {
      keys.push(mainKey.toUpperCase());
    } else {
      keys.push(mainKey); // F1, Escape, etc.
    }
  }

  // Format as hotkey string
  if (keys.length >= 2) {
    const hotkeyString = keys.join("+");
    setCapturedHotkey(hotkeyString);
  }
};
Save Function:
const saveHotkey = async () => {
  if (!editingHotkey || !capturedHotkey) return;

  // Update local state
  const updatedHotkeys = {
    ...hotkeys,
    [editingHotkey]: capturedHotkey,
  };
  setHotkeys(updatedHotkeys);

  // Save to persistent storage
  await saveSettings({ hotkeys: updatedHotkeys });

  // Re-register hotkeys with new configuration
  const settings = await loadSettings();
  await invoke("register_hotkeys", { hotkeyConfig: settings });

  // Check for conflicts
  const conflictsResult = await invoke("check_hotkey_conflicts", {
    hotkeyConfig: settings,
  });

  if (conflictsResult.has_conflicts) {
    setHotkeyConflicts(conflictsResult.conflicts);
  } else {
    setHotkeyConflicts({});
  }

  // Exit edit mode
  cancelEditingHotkey();
};
UI Component (Settings Modal):
// Display each hotkey with edit button
{Object.entries(hotkeys).map(([hotkeyName, hotkeyValue]) => (
  <div key={hotkeyName}>
    <span>{formatHotkeyName(hotkeyName)}</span>
    {editingHotkey === hotkeyName ? (
      // Edit mode: capture keyboard input
      <div
        tabIndex={0}
        onKeyDown={handleHotkeyCapture}
        autoFocus
      >
        <code>{capturedHotkey || "Press keys..."}</code>
        <button onClick={saveHotkey}>✓</button>
        <button onClick={cancelEditingHotkey}>✕</button>
      </div>
    ) : (
      // View mode: show hotkey with edit button
      <div>
        <code>{hotkeyValue}</code>
        <button onClick={() => startEditingHotkey(hotkeyName)}>
          Edit
        </button>
      </div>
    )}
  </div>
))}

Default Hotkeys

The following hotkeys are configured by default:

  • Toggle Drawing Mode: Ctrl+Shift+D
  • Arrow Tool: Ctrl+Shift+A
  • Circle Tool: Ctrl+Shift+C
  • Box Tool: Ctrl+Shift+B
  • Freehand Tool: Ctrl+Shift+F
  • Highlighter Tool: Ctrl+Shift+H
  • Text Tool: Ctrl+Shift+T

All of these can be changed through the Settings panel!
