2. Capture Key Combinations

Key Capture Handler

src/components/MiniPanel.tsx:361-394
/** * Feature #64: Handle keyboard input during hotkey capture * Captures the key combination and formats it as a hotkey string */ const handleHotkeyCapture = (e: React.KeyboardEvent) => { if (!editingHotkey) return; e.preventDefault(); const keys: string[] = []; // Capture modifier keys if (e.ctrlKey) keys.push("Ctrl"); if (e.shiftKey) keys.push("Shift"); if (e.altKey) keys.push("Alt"); if (e.metaKey || e.key === "Meta") keys.push(isMac() ? "Cmd" : "Win"); // Capture the main key const mainKey = e.key; if ( !["Control", "Shift", "Alt", "Meta"].includes(mainKey) && mainKey.length === 1 ) { // Single character key (letter, number, symbol) keys.push(mainKey.toUpperCase()); } else if ( !["Control", "Shift", "Alt", "Meta"].includes(mainKey) && mainKey.length > 1 ) { // Special key (F1, Escape, etc.) keys.push(mainKey); } // Format as hotkey string (e.g., "Ctrl+Shift+A") if (keys.length >= 2) { const hotkeyString = keys.join("+"); setCapturedHotkey(hotkeyString); } };

Key Features of Capture Handler:

  • Captures all modifiers: Ctrl, Shift, Alt, Meta/Cmd
  • Captures main key: Letters, numbers, special keys (F1-F12, Escape, etc.)
  • Platform-aware: Displays "Cmd" on Mac, "Win" on Windows/Linux
  • Formats output: Joins with "+" (e.g., "Ctrl+Shift+A")
  • Validates input: Requires at least 2 keys (modifier + main key)
  • Real-time feedback: Updates UI immediately as keys are pressed

/** * Feature #64: Save the edited hotkey * Validates the hotkey, checks for conflicts, and saves to storage */ const saveHotkey = async () => { if (!editingHotkey || !capturedHotkey) return; try { // 1. Update local state const updatedHotkeys = { ...hotkeys, [editingHotkey]: capturedHotkey, }; setHotkeys(updatedHotkeys); // 2. Save to persistent storage await saveSettings({ hotkeys: updatedHotkeys as any }); console.log(`Updated hotkey ${editingHotkey} to: ${capturedHotkey}`); // 3. Re-register hotkeys with new configuration const settings = await loadSettings(); await invoke("register_hotkeys", { hotkeyConfig: settings }); console.log("Hotkeys re-registered with new configuration"); // 4. Check for conflicts with the new hotkey const conflictsResult = await invoke<Record<string, any>>("check_hotkey_conflicts", { hotkeyConfig: settings, }); if (conflictsResult.has_conflicts) { setHotkeyConflicts(conflictsResult.conflicts); console.warn("Hotkey conflicts detected after update:", conflictsResult.conflicts); } else { // Clear conflicts if none exist setHotkeyConflicts({}); } // 5. Exit edit mode cancelEditingHotkey(); } catch (error) { console.error("Failed to save hotkey:", error); } };

Validation and Saving Process:

  • Pre-flight check: Ensures editingHotkey and capturedHotkey are set
  • Update state: Merges new hotkey into existing configuration
  • Persist storage: Saves to Tauri's persistent store
  • Re-register: Updates global shortcuts with new configuration
  • Conflict detection: Checks for system-level hotkey conflicts
  • User feedback: Shows warnings if conflicts detected
  • Cleanup: Exits recording mode after successful save
