// Feature #105: Get tool-specific color from settings const getToolColor = useCallback((tool: ToolType | null): string => { if (!tool || !settings) return defaultColor; // Map tool type to color setting key const toolColorKey: Record<ToolType, keyof Settings['colors']> = { [ToolType.ARROW]: 'arrow', [ToolType.CIRCLE]: 'circle', [ToolType.BOX]: 'box', [ToolType.FREEHAND]: 'freehand', [ToolType.HIGHLIGHTER]: 'highlighter', [ToolType.TEXT]: 'text', }; const colorKey = toolColorKey[tool]; return settings.colors[colorKey] || defaultColor; }, [settings, defaultColor]);

How It Works

  1. On first launch, DEFAULT_SETTINGS are saved to persistent storage
  2. Settings are loaded from storage on app startup
  3. When a tool is selected, getToolColor() and getToolLineThickness() retrieve values
  4. When a shape is created, these values are applied to the shape
  5. Users can customize values via settings panel (Features #105, #106)
  6. Customized values persist across app restarts
