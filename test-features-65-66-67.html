    #[tauri::command]
fn unregister_all_hotkeys(app: AppHandle) -> Result<(), String> {
    // Get the registry of registered hotkeys
    if let Some(registry) = app.state::().try_get() {
        let mut registry_guard = registry.lock()
            .map_err(|e| format!("Registry lock error: {}", e))?;

        // Unregister each hotkey
        for hotkey_str in registry_guard.registered_hotkeys.iter() {
            if let Ok(shortcut) = parse_hotkey_string(hotkey_str) {
                app.global_shortcut().unregister(shortcut)?;
                println!("Unregistered hotkey: {}", hotkey_str);
            }
        }

        // Clear the registry
        registry_guard.registered_hotkeys.clear();
        println!("All hotkeys unregistered");
    }
    Ok(())
}

How It Works

  1. Registration: When register_hotkeys is called, it first unregisters all existing hotkeys
  2. Tracking: Each successfully registered hotkey is added to the registry
  3. Deregistration: Before registering new hotkeys, all existing ones are unregistered
  4. Dynamic Updates: When user changes hotkey settings, old ones are cleared before new ones are added
